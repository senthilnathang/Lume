// backend/src/domains/ticket/ticket.entity.ts

import { defineEntity, defineField } from '@lume/runtime'

export const Ticket = defineEntity({
  // Metadata
  slug: 'ticket',
  name: 'Ticket',
  label: 'Support Ticket',
  description: 'Customer support tickets',
  icon: 'ticket',
  color: '#2196F3',

  // ORM
  orm: 'drizzle',
  tableName: 'tickets',
  softDelete: true,
  auditable: true,

  // Fields
  fields: [
    defineField('title', 'text', {
      label: 'Title',
      required: true,
      validation: [{ rule: 'minLength', value: 5 }]
    }),
    defineField('description', 'rich-text', {
      label: 'Description',
      required: true
    }),
    defineField('status', 'select', {
      label: 'Status',
      required: true,
      defaultValue: 'open',
      validation: [{ rule: 'enum', values: ['open', 'in_progress', 'waiting', 'closed'] }]
    }),
    defineField('priority', 'select', {
      label: 'Priority',
      defaultValue: 'normal',
      validation: [{ rule: 'enum', values: ['low', 'normal', 'high', 'urgent'] }]
    }),
    defineField('category', 'select', {
      label: 'Category',
      validation: [{ rule: 'enum', values: ['bug', 'feature', 'question', 'other'] }]
    }),
    defineField('assignedTo', 'relation', {
      label: 'Assigned To',
      relation: 'User',
      relationType: 'many-to-one'
    }),
    defineField('customer', 'relation', {
      label: 'Customer',
      relation: 'Customer',
      required: true,
      relationType: 'many-to-one'
    }),
    defineField('resolution', 'rich-text', {
      label: 'Resolution',
      computed: false  // Shown only when status='closed'
    }),
    defineField('daysOpen', 'number', {
      label: 'Days Open',
      computed: true,
      computed_expression: 'Math.floor((Date.now() - this.createdAt) / 86400000)'
    }),
    defineField('isOverdue', 'boolean', {
      label: 'Overdue',
      computed: true,
      computed_expression: 'this.daysOpen > 2 && this.status !== "closed"'
    }),
    defineField('slaStatus', 'text', {
      label: 'SLA Status',
      computed: true,
      computed_expression: 'this.isOverdue ? "breached" : "on_track"'
    }),
  ],

  // Relationships
  relations: [
    { name: 'customer', target: 'Customer', type: 'many-to-one', required: true },
    { name: 'assignedTo', target: 'User', type: 'many-to-one' },
    { name: 'comments', target: 'TicketComment', type: 'one-to-many' },
  ],

  // Hooks
  hooks: {
    beforeCreate: [
      // Auto-assign to available rep
      async (record, context) => {
        const availableRep = await findAvailableRepByCategory(record.category)
        if (availableRep) {
          record.assignedTo = availableRep.id
        }
      },
      // Auto-set priority based on category
      async (record) => {
        if (record.category === 'bug') record.priority = 'high'
      }
    ],
    afterCreate: [
      async (record, context) => {
        // Log to audit trail
        console.log(`Ticket created: ${record.id}`)
      }
    ],
    beforeUpdate: [
      async (record, context) => {
        // Validate workflow state
        if (record.status === 'closed' && !record.resolution) {
          throw new Error('Resolution required before closing')
        }
      }
    ],
    afterUpdate: [
      async (record, context) => {
        // Notify if assigned rep changed
        if (context.changedFields.includes('assignedTo')) {
          // Trigger notification workflow
        }
      }
    ]
  },

  // Workflows triggered by entity events
  workflows: {
    onCreate: [
      'ticket.on_create.notify_customer',
      'ticket.on_create.notify_assigned_rep',
      'ticket.on_create.create_slack_message'
    ],
    onUpdate: [
      'ticket.on_update.notify_if_status_changed',
      'ticket.on_update.notify_if_escalated'
    ],
    onDelete: []
  },

  // Agents (reactive logic)
  agents: [
    {
      id: 'check_overdue',
      trigger: 'status != "closed" && daysOpen > 2',
      schedule: '*/5 * * * *',  // Every 5 minutes
      action: {
        type: 'escalate',
        updates: {
          priority: '"urgent"',
          notes: 'concat(notes, "\\nAuto-escalated: SLA breached")'
        },
        notify: 'ticket.manager@company.com'
      }
    },
    {
      id: 'auto_close_resolved',
      trigger: 'status = "waiting" && daysOpen > 7',
      schedule: '0 2 * * *',  // Daily at 2 AM
      action: {
        type: 'workflow',
        workflowId: 'ticket.auto_close.send_survey'
      }
    }
  ],

  // Permissions (RBAC + ABAC)
  permissions: [
    {
      resource: 'ticket',
      action: 'create',
      rule: 'user.role = "support_agent" OR user.role = "manager"',
      allow: true
    },
    {
      resource: 'ticket',
      action: 'read',
      rule: 'user.role = "manager" OR assignedTo = user.id OR customer.accountId = user.accountId',
      scope: 'return records matching condition'
    },
    {
      resource: 'ticket',
      action: 'update',
      rule: 'user.role = "manager" OR assignedTo = user.id',
      fieldLevel: {
        'status': 'user.role = "manager"',
        'priority': 'user.role = "manager"',
        'resolution': 'user.role = "support_agent" OR assignedTo = user.id',
        'notes': 'true'  // Anyone can add notes
      }
    },
    {
      resource: 'ticket',
      action: 'delete',
      rule: 'user.role = "admin"',
      allow: true
    }
  ],

  // Views (auto-generated + custom)
  views: [
    {
      id: 'table',
      type: 'table',
      label: 'All Tickets',
      columns: ['title', 'status', 'priority', 'assignedTo', 'createdAt'],
      filters: ['status', 'priority', 'assignedTo'],
      sorting: { defaultBy: 'createdAt', order: 'desc' },
      inlineEdit: true
    },
    {
      id: 'kanban',
      type: 'kanban',
      label: 'Status Board',
      groupBy: 'status',
      columns: ['title', 'priority', 'assignedTo'],
      dragToUpdate: { field: 'status' }
    },
    {
      id: 'calendar',
      type: 'calendar',
      label: 'Timeline',
      dateField: 'createdAt',
      groupBy: 'assignedTo',
      colors: { priority: true }
    },
    {
      id: 'my_tickets',
      type: 'table',
      label: 'My Tickets',
      defaultFilter: { assignedTo: '${user.id}' },
      columns: ['title', 'status', 'daysOpen', 'isOverdue']
    }
  ]
})

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { runtime } from '@lume/runtime'
import type { TicketRecord } from '@/domains/ticket/ticket.entity'

const entity = runtime.getEntityMetadata('ticket')
const records = ref<TicketRecord[]>([])
const activeView = ref('table')
const loading = ref(false)
const drawerVisible = ref(false)
const editingRecord = ref<TicketRecord | null>(null)

// Computed: available views (filtered by permission)
const availableViews = computed(() => {
  return entity.views.filter(v => user.can('ticket.read'))
})

// Computed: visible columns (filtered by permission)
const tableColumns = computed(() => {
  return entity.fields
    .filter(f => user.can('ticket.read', f.name))
    .map(f => ({
      title: f.label,
      dataIndex: f.name,
      sortable: true,
      filterable: entity.views[0].filters?.includes(f.name)
    }))
})

// Computed: visible fields in form (filtered by permission)
const visibleFields = computed(() => {
  return entity.fields.filter(f => {
    const action = editingRecord.value ? 'update' : 'create'
    return user.can(`ticket.${action}`, f.name)
  })
})

// Load data
async function loadViewData() {
  loading.value = true
  try {
    const result = await runtime.execute({
      context: { user },
      entity: 'ticket',
      action: 'read',
      options: { view: activeView.value }
    })
    records.value = result.data
  } finally {
    loading.value = false
  }
}

// Save field (inline edit)
async function saveField(record: TicketRecord, fieldName: string) {
  const permission = user.can('ticket.update', fieldName)
  if (!permission) {
    throw new Error('You don\'t have permission to edit this field')
  }

  await runtime.execute({
    context: { user },
    entity: 'ticket',
    action: 'update',
    data: { id: record.id, [fieldName]: record[fieldName] }
  })
}

// Save ticket (form)
async function saveTicket() {
  const action = editingRecord.value ? 'update' : 'create'
  const result = await runtime.execute({
    context: { user },
    entity: 'ticket',
    action,
    data: editingRecord.value
  })

  if (result.success) {
    await loadViewData()
    drawerVisible.value = false
  }
}

onMounted(() => loadViewData())
</script>

What This Example Demonstrates

  • Zero-code app generation: Define entity once, get working API + UI automatically
  • Computed fields: daysOpen, isOverdue, slaStatus calculated on demand
  • Hooks: Auto-assign, validation, notifications at lifecycle events
  • Workflows: Complex async logic (emails, Slack, retry logic) defined visually
  • Agents: Scheduled jobs that react to conditions (overdue escalation, auto-close)
  • Permissions: Field-level access control (who can edit status, priority, resolution)
  • Views: Table, Kanban, Calendar auto-generated from entity definition
  • Full extensibility: All code is accessible and customizable
